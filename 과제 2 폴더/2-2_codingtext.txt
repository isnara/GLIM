#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> board, int k) {
    int answer = 0;
    
    for (int i = 0; i < board.size() && i < 100; ++i)
    {
        vector<int>& vecCol = board[i];
        for (int j = 0; j < vecCol.size() && j < 100; ++j)
        {
            if (i + j <= k)
                answer += vecCol[j];
		}
    }
    
    return answer;
}
